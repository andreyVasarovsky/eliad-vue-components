Installation:<br>
npm install @eliad/vue-components<br><br>

Usage:<br>
import {createApp} from 'vue'; <br>
import av from "@eliad/vue-components" <br>
import '@eliad/vue-components/dist/css/av.css' <br><br>

const app = createApp({});<br>
app.use(av).mount('#app');<br>
